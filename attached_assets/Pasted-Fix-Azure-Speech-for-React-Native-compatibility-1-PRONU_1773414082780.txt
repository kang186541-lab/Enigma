1. PRONUNCIATION ASSESSMENT:
Replace Web Speech API with Azure Speech REST API:

const assessPronunciation = async (audioUri, referenceText, language) => {
  try {
    // Read audio file as base64
    const audioBase64 = await FileSystem.readAsStringAsync(audioUri, {
      encoding: FileSystem.EncodingType.Base64
    })

    // Language codes
    const langCode = {
      english: 'en-US',
      korean: 'ko-KR',
      spanish: 'es-ES'
    }[language] || 'en-US'

    // Pronunciation assessment config
    const pronunciationConfig = {
      ReferenceText: referenceText,
      GradingSystem: 'HundredMark',
      Granularity: 'Phoneme',
      EnableMiscue: true
    }
    const configBase64 = btoa(JSON.stringify(pronunciationConfig))

    const response = await fetch(
      `https://koreacentral.stt.speech.microsoft.com/speech/recognition/conversation/cognitiveservices/v1?language=${langCode}`,
      {
        method: 'POST',
        headers: {
          'Ocp-Apim-Subscription-Key': AZURE_SPEECH_KEY,
          'Content-Type': 'audio/wav',
          'Pronunciation-Assessment': configBase64
        },
        body: Buffer.from(audioBase64, 'base64')
      }
    )

    const result = await response.json()
    return {
      score: result.NBest?.[0]?.PronunciationAssessment?.PronScore || 0,
      feedback: result.NBest?.[0]?.PronunciationAssessment
    }
  } catch (error) {
    console.error('발음 평가 오류:', error)
    return { score: 0, feedback: null }
  }
}

4. AZURE TTS for React Native:
Replace browser Audio with expo-av:

import { Audio } from 'expo-av'

const playAzureTTS = async (text, voice) => {
  try {
    const response = await fetch(
      'https://koreacentral.tts.speech.microsoft.com/cognitiveservices/v1',
      {
        method: 'POST',
        headers: {
          'Ocp-Apim-Subscription-Key': AZURE_SPEECH_KEY,
          'Content-Type': 'application/ssml+xml',
          'X-Microsoft-OutputFormat': 'audio-16khz-128kbitrate-mono-mp3'
        },
        body: `<speak version='1.0' xml:lang='ko-KR'>
          <voice name='${voice}'>${text}</voice>
        </speak>`
      }
    )

    const audioBlob = await response.blob()
    const reader = new FileReader()
    reader.readAsDataURL(audioBlob)
    reader.onloadend = async () => {
      const base64Audio = reader.result.split(',')[1]
      const fileUri = FileSystem.documentDirectory + 'tts_audio.mp3'
      await FileSystem.writeAsStringAsync(fileUri, base64Audio, {
        encoding: FileSystem.EncodingType.Base64
      })
      const { sound } = await Audio.Sound.createAsync({ uri: fileUri })
      await sound.playAsync()
    }
  } catch (error) {
    console.error('TTS 오류:', error)
  }
}
